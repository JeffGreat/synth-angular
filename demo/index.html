        $scope.books = [{
            title: "Manon Lescaut",
            author: "Abbé Prévost"
        }, {
            title: "Le Compte de Monte Cristo",
            author: "Dumas"
        }];
        
        $scope.getContent = function(item){
            return item.author.toUpperCase();
        };

            title: 'My list',
            fields: [{
                property: 'title',

                expression: $scope.getContent,
